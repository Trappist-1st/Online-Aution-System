## 第二步：定义公共接口

### 2.1 创建 AuctionService 接口

**文件**: `src/main/java/com/auction/common/AuctionService.java`

```java
package com.auction.common;

import com.auction.server.AuctionItem;
import java.rmi.Remote;
import java.rmi.RemoteException;

/**
 * 拍卖服务远程接口
 * 定义服务器端提供的远程方法
 */
public interface AuctionService extends Remote {
    
    /**
     * 获取当前拍卖物品信息
     * @return 拍卖物品对象
     * @throws RemoteException RMI 异常
     */
    AuctionItem getAuctionItem() throws RemoteException;
    
    /**
     * 客户端提交出价
     * @param bidderName 出价者姓名
     * @param amount 出价金额
     * @return true 表示出价成功，false 表示出价失败
     * @throws RemoteException RMI 异常
     */
    boolean placeBid(String bidderName, double amount) throws RemoteException;
    
    /**
     * 注册客户端回调接口
     * @param callback 客户端回调对象
     * @throws RemoteException RMI 异常
     */
    void registerClient(ClientCallback callback) throws RemoteException;
    
    /**
     * 注销客户端回调接口
     * @param callback 客户端回调对象
     * @throws RemoteException RMI 异常
     */
    void unregisterClient(ClientCallback callback) throws RemoteException;
}
```

```java
package com.auction.common;

import java.rmi.Remote;
import java.rmi.RemoteException;

/**
 * 客户端回调接口
 * 用于服务器向客户端推送更新通知
 */
public interface ClientCallback extends Remote {
    
    /**
     * 通知客户端出价更新
     * @param itemName 物品名称
     * @param currentPrice 当前最高价
     * @param currentBidder 当前最高出价者
     * @throws RemoteException RMI 异常
     */

                        String currentBidder) throws RemoteException;
}
```

## 第三步：创建数据模型

### 3.1 创建 AuctionItem 类

**文件**: `src/main/java/com/auction/server/AuctionItem.java`

```java
package com.auction.server;

import java.io.Serializable;

/**
 * 拍卖物品数据模型
 * 必须实现 Serializable 以便通过 RMI 传输
 */
public class AuctionItem implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private String itemName;        // 物品名称
    private double startingPrice;   // 起拍价
    private double currentPrice;    // 当前最高价
    private String currentBidder;   // 当前最高出价者
    private boolean isActive;       // 拍卖是否进行中
    
    /**
     * 构造函数
     * @param itemName 物品名称
     * @param startingPrice 起拍价
     */
    public AuctionItem(String itemName, double startingPrice) {
        this.itemName = itemName;
        this.startingPrice = startingPrice;
        this.currentPrice = startingPrice;
        this.currentBidder = "无";
        this.isActive = true;
    }
    
    // Getter 和 Setter 方法
    public String getItemName() {
        return itemName;
    }
    
    public double getStartingPrice() {
        return startingPrice;
    }
    
    public double getCurrentPrice() {
        return currentPrice;
    }
    
    public void setCurrentPrice(double currentPrice) {
        this.currentPrice = currentPrice;
    }
    
    public String getCurrentBidder() {
        return currentBidder;
    }
    
    public void setCurrentBidder(String currentBidder) {
        this.currentBidder = currentBidder;
    }
    
    public boolean isActive() {
        return isActive;
    }
    
    public void setActive(boolean active) {
        isActive = active;
    }
    
    @Override
    public String toString() {
        return "物品: " + itemName + 
               ", 起拍价: " + startingPrice + 
               ", 当前价: " + currentPrice + 
               ", 出价者: " + currentBidder;
    }
}
```

## 第四步：实现服务器端

### 4.1 创建 AuctionServiceImpl 类

**文件**: `src/main/java/com/auction/server/AuctionServiceImpl.java`

```java
package com.auction.server;

import com.auction.common.AuctionService;
import com.auction.common.ClientCallback;
import java.rmi.RemoteException;
import java.rmi.server.UnicastRemoteObject;
import java.util.ArrayList;
import java.util.List;

/**
 * 拍卖服务实现类
 * 实现核心拍卖业务逻辑
 */
public class AuctionServiceImpl extends UnicastRemoteObject implements AuctionService {
    
    private AuctionItem auctionItem;                    // 拍卖物品
    private List<ClientCallback> clients;               // 已注册的客户端列表
    
    /**
     * 构造函数
     * @param itemName 拍卖物品名称
     * @param startingPrice 起拍价
     * @throws RemoteException RMI 异常
     */
    public AuctionServiceImpl(String itemName, double startingPrice) throws RemoteException {
        super();
        this.auctionItem = new AuctionItem(itemName, startingPrice);
        this.clients = new ArrayList<>();
        System.out.println("拍卖服务已初始化");
        System.out.println("物品: " + itemName + ", 起拍价: " + startingPrice + " 元");
    }
    
    @Override
    public synchronized AuctionItem getAuctionItem() throws RemoteException {
        return auctionItem;
    }
    
    @Override
    public synchronized boolean placeBid(String bidderName, double amount) throws RemoteException {
        // 验证出价
        if (amount <= auctionItem.getCurrentPrice()) {
            System.out.println("❌ 无效出价: " + bidderName + " 出价 " + amount + 
                             " 元（当前最高价: " + auctionItem.getCurrentPrice() + " 元）");
            return false;
        }
        
        // 更新拍卖信息
        auctionItem.setCurrentPrice(amount);
        auctionItem.setCurrentBidder(bidderName);
        
        System.out.println("✅ 新的最高价: " + amount + " 元，出价者: " + bidderName);
        
        // 通知所有客户端
        notifyAllClients();
        
        return true;
    }
    
    @Override
    public synchronized void registerClient(ClientCallback callback) throws RemoteException {
        if (!clients.contains(callback)) {
            clients.add(callback);
            System.out.println("客户端已注册，当前在线客户端数: " + clients.size());
        }
    }
    
    @Override
    public synchronized void unregisterClient(ClientCallback callback) throws RemoteException {
        clients.remove(callback);
        System.out.println("客户端已注销，当前在线客户端数: " + clients.size());
    }
    
    /**
     * 通知所有已注册的客户端
     */
    private void notifyAllClients() {
        List<ClientCallback> failedClients = new ArrayList<>();
        
        for (ClientCallback client : clients) {
            try {
                client.notifyBidUpdate(
                    auctionItem.getItemName(),
                    auctionItem.getCurrentPrice(),
                    auctionItem.getCurrentBidder()
                );
            } catch (RemoteException e) {
                System.out.println("⚠️ 通知客户端失败，将移除该客户端");
                failedClients.add(client);
            }
        }
        
        // 移除失败的客户端
        clients.removeAll(failedClients);
    }
}
```

### 4.2 创建 AuctionServer 启动类

**文件**: `src/main/java/com/auction/server/AuctionServer.java`

```java
package com.auction.server;

import java.rmi.registry.LocateRegistry;
import java.rmi.registry.Registry;

/**
 * 拍卖服务器启动类
 */
public class AuctionServer {
    
    private static final int RMI_PORT = 1099;
    private static final String SERVICE_NAME = "AuctionService";
    
    public static void main(String[] args) {
        try {
            System.out.println("========================================");
            System.out.println("      拍卖服务器启动中...");
            System.out.println("========================================");
            
            // 创建并启动 RMI Registry
            Registry registry = LocateRegistry.createRegistry(RMI_PORT);
            System.out.println("✅ RMI Registry 已在端口 " + RMI_PORT + " 上启动");
            
            // 创建拍卖服务实例
            AuctionServiceImpl auctionService = new AuctionServiceImpl("明代青花瓷", 10000.0);
            
            // 将服务绑定到 RMI Registry
            registry.rebind(SERVICE_NAME, auctionService);
            System.out.println("✅ 拍卖服务已绑定到 RMI Registry (名称: " + SERVICE_NAME + ")");
            
            System.out.println("========================================");
            System.out.println("      服务器已就绪，等待客户端连接...");
            System.out.println("========================================\n");
            
            // 保持服务器运行
            Thread.sleep(Long.MAX_VALUE);
            
        } catch (Exception e) {
            System.err.println("❌ 服务器启动失败: " + e.getMessage());
            e.printStackTrace();
        }
    }
}
```

### 5.2 创建 AuctionClient 主类

**文件**: `src/main/java/com/auction/client/AuctionClient.java`

```java
package com.auction.client;

import com.auction.common.AuctionService;
import com.auction.common.ClientCallback;
import com.auction.server.AuctionItem;
import java.rmi.registry.LocateRegistry;
import java.rmi.registry.Registry;
import java.util.Scanner;

/**
 * 拍卖客户端主类
 */
public class AuctionClient {
    
    private static final String SERVER_HOST = "localhost";
    private static final int RMI_PORT = 1099;
    private static final String SERVICE_NAME = "AuctionService";
    
    private AuctionService auctionService;
    private ClientCallback callback;
    private Scanner scanner;
    
    public AuctionClient() {
        this.scanner = new Scanner(System.in);
    }
    
    /**
     * 连接到服务器
     */
    public void connect() {
        try {
            System.out.println("正在连接到拍卖服务器...");
            
            // 获取 RMI Registry
            Registry registry = LocateRegistry.getRegistry(SERVER_HOST, RMI_PORT);
            
            // 查找拍卖服务
            auctionService = (AuctionService) registry.lookup(SERVICE_NAME);
            
            // 创建回调对象并注册
            callback = new ClientCallbackImpl();
            auctionService.registerClient(callback);
            
            System.out.println("✅ 成功连接到拍卖服务器！\n");
            
        } catch (Exception e) {
            System.err.println("❌ 连接失败: " + e.getMessage());
            System.exit(1);
        }
    }
    
    /**
     * 显示当前拍卖信息
     */
    public void displayAuctionInfo() {
        try {
            AuctionItem item = auctionService.getAuctionItem();
            System.out.println("\n========================================");
            System.out.println("          当前拍卖信息");
            System.out.println("========================================");
            System.out.println("物品名称: " + item.getItemName());
            System.out.println("起拍价: " + item.getStartingPrice() + " 元");
            System.out.println("当前最高价: " + item.getCurrentPrice() + " 元");
            System.out.println("当前最高出价者: " + item.getCurrentBidder());
            System.out.println("========================================\n");
        } catch (Exception e) {
            System.err.println("❌ 获取拍卖信息失败: " + e.getMessage());
        }
    }
    
    /**
     * 提交出价
     */
    public void placeBid() {
        try {
            System.out.print("请输入您的姓名: ");
            String name = scanner.nextLine().trim();
            
            if (name.isEmpty()) {
                System.out.println("❌ 姓名不能为空！");
                return;
            }
            
            System.out.print("请输入出价金额: ");
            String amountStr = scanner.nextLine().trim();
            
            double amount;
            try {
                amount = Double.parseDouble(amountStr);
            } catch (NumberFormatException e) {
                System.out.println("❌ 无效的金额格式！");
                return;
            }
            
            if (amount <= 0) {
                System.out.println("❌ 出价必须大于 0！");
                return;
            }
            
            // 提交出价
            boolean success = auctionService.placeBid(name, amount);
            
            if (success) {
                System.out.println("✅ 出价成功！");
            } else {
                AuctionItem item = auctionService.getAuctionItem();
                System.out.println("❌ 出价失败！您的出价必须高于当前最高价: " + 
                                 item.getCurrentPrice() + " 元");
            }
            
        } catch (Exception e) {
            System.err.println("❌ 出价失败: " + e.getMessage());
        }
    }
    
    /**
     * 运行客户端主循环
     */
    public void run() {
        displayAuctionInfo();
        
        while (true) {
            System.out.println("\n请选择操作:");
            System.out.println("1 - 出价");
            System.out.println("2 - 查看当前拍卖信息");
            System.out.println("0 - 退出");
            System.out.print("请输入选项: ");
            
            String choice = scanner.nextLine().trim();
            
            switch (choice) {
                case "1":
                    placeBid();
                    break;
                case "2":
                    displayAuctionInfo();
                    break;
                case "0":
                    disconnect();
                    return;
                default:
                    System.out.println("❌ 无效的选项，请重新输入！");
            }
        }
    }
    
    /**
     * 断开连接
     */
    public void disconnect() {
        try {
            if (callback != null && auctionService != null) {
                auctionService.unregisterClient(callback);
            }
            System.out.println("\n已退出拍卖系统，再见！");
        } catch (Exception e) {
            System.err.println("❌ 断开连接时出错: " + e.getMessage());
        }
    }
    
    /**
     * 主方法
     */
    public static void main(String[] args) {
        System.out.println("========================================");
        System.out.println("      欢迎使用在线拍卖系统");
        System.out.println("========================================\n");
        
        AuctionClient client = new AuctionClient();
        client.connect();
        client.run();
    }
}
```

## 常见问题排查

### 问题 1: 编译错误 - 找不到符号

**解决方法**：
- 确认目录结构正确
- 确认包名与目录结构一致
- 使用 `-cp` 指定正确的类路径

### 问题 2: 运行时 ClassNotFoundException

**解决方法**：
```bash
# 确保使用正确的类路径
java -cp bin com.auction.server.AuctionServer
```

### 问题 3: java.rmi.ConnectException

**解决方法**：
- 确认服务器已启动
- 检查服务器 IP 和端口
- 关闭防火墙或添加例外

### 问题 4: 中文乱码

**解决方法**：
```bash
# 编译时指定编码
javac -encoding UTF-8 ...

# 运行时指定编码
java -Dfile.encoding=UTF-8 -cp bin com.auction.server.AuctionServer
```
